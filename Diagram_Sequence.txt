sequenceDiagram
    actor Tourist
    participant GPS as Background GPS
    participant Scanner as QR Scanner
    participant App as Mobile App
    participant Geofence as Geofence Engine
    participant Narration as Narration Engine
    participant Map as Goong Maps
    participant API as Backend REST API (SQL Server)
    
    %% Startup & Data Load
    App->>API: GET /api/pois (District 4)
    API-->>App: List[POI] Data + Radiuses
    App->>Map: WebView.Source = BuildMapHtml()
    App->>GPS: Start Tracking Service
    
    %% Main GPS Loop
    loop Every 5 Seconds
        GPS-->>App: Location (Lat, Lng)
        App->>Map: Update Blue Dot (Eval)
        
        alt User moved > 5m
            App->>Geofence: CheckGeofences(location, List[POI])
            Geofence->>Geofence: Sort by Priority
            
            alt Inside Radius & Cooldown Passed
                Geofence->>App: PoiTriggeredEvent(POI)
                App->>Map: Highlight Marker (Eval)
                App->>App: Show POI Bottom Sheet
                
                %% Audio Decision
                App->>API: GET /api/audio?poiId=X&lang=vi
                alt Configured Audio File Exists
                    API-->>App: MP3 File URL
                    App->>Narration: Play Natural Offline Audio
                else No Audio File Available
                    API-->>App: Text Script Only
                    App->>Narration: SpeakAsync(Text, vi-VN)
                end
                
                Narration-->>Tourist: Audio Instructions
                App->>API: POST /api/narration-log (Type: Geofence)
            end
        end
    end
    
    %% Manual QR Code Bypass
    Tourist->>Scanner: Scan QR Code at Bus Stop
    Scanner-->>App: Discovered URL / POI ID
    App->>API: GET /api/audio?poiId=X
    API-->>App: Text or MP3 Info
    App->>Narration: Clear Queue & Force Play
    Narration-->>Tourist: Instant Audio (Bypass GPS)
    App->>API: POST /api/narration-log (Type: QRCode)
